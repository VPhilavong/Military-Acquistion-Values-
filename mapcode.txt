mapdata <- map_data("world") %>% rename(country = region)
eda_country <- eda_data %>%
  filter(Implementing_Agency %in% c('Navy', 'Army'), Fiscal_Year_of_Request >= 2010 , Fiscal_Year_of_Request <= 2020) %>%
  rename(year = Fiscal_Year_of_Request, total = Total_Acquisition_Value, country = Country) %>%
  group_by(country, Implementing_Agency) %>%
  summarise(average = mean(total, na.rm = TRUE))

# Join and maintain all countries
mapdata_navy <- left_join(mapdata, eda_country %>% filter(Implementing_Agency == 'Navy'), by = "country")
mapdata_army <- left_join(mapdata, eda_country %>% filter(Implementing_Agency == 'Army'), by = "country")

map1 <- mapdata_navy %>% ggplot(aes(x = long, y = lat, group = group, fill = average)) +
  geom_polygon(color = "black") + 
  theme_minimal() + 
 scale_fill_gradient(low = "blue", high = "red", labels = scales::label_dollar(prefix = "$")) +
  labs(fill = "Mean Acquisition Value", title = "Average Acquisition of the Navy from 2010 to 2020")

map2 <- mapdata_army %>% ggplot(aes(x = long, y = lat, group = group, fill = average)) +
  geom_polygon(color = "black") + 
  theme_minimal() + 
  scale_fill_gradient(low = "blue", high = "red", labels = scales::label_dollar(prefix = "$")) +
  labs(fill = "Mean Acquisition Value", title = "Average Acquisition of the Army from 2010 to 2020")
